Here is the equivalent CLI script:\n\n```bash\n#!/bin/bash\n\n# Define parameters\nvmSku="Standard_A1_v2"\nubuntuOSVersion="18.04-LTS"\nvmssName="zytest"\ninstanceCount="zytest"\nadminUsername="zytest"\nlocation="westus"\nauthenticationType="sshPublicKey"\nadminPasswordOrKey="zytest"\n\n# Create Network Security Group\naz network nsg create --name ${vmssName}nsg --resource-group myResourceGroup --location $location\n\n# Create Virtual Network\naz network vnet create --name ${vmssName}vnet --resource-group myResourceGroup --location $location --address-prefix 10.0.0.0/16 --subnet-name ${vmssName}subnet --subnet-prefix 10.0.0.0/24 --network-security-group ${vmssName}nsg\n\n# Create Storage Account\naz storage account create --name ${vmssName}sa --resource-group myResourceGroup --location $location --sku Standard_LRS --kind StorageV2\n\n# Create Public IP Address\naz network public-ip create --name ${vmssName}pip --resource-group myResourceGroup --location $location --dns-name ${vmssName} --allocation-method Dynamic\n\n# Create Network Interface\naz network nic create --name ${vmssName}nic --resource-group myResourceGroup --location $location --subnet ${vmssName}subnet --vnet-name ${vmssName}vnet --public-ip-address ${vmssName}pip\n\n# Create Virtual Machine\naz vm create --name ${vmssName}vm --resource-group myResourceGroup --location $location --nics ${vmssName}nic --image UbuntuLTS --admin-username $adminUsername --admin-password $adminPasswordOrKey --size $vmSku --storage-account ${vmssName}sa\n\n# Create Load Balancer\naz network lb create --name ${vmssName}lb --resource-group myResourceGroup --location $location --frontend-ip-name LoadBalancerFrontEnd --backend-pool-name ${vmssName}bepool --subnet ${vmssName}subnet --vnet-name ${vmssName}vnet\n\n# Create Virtual Machine Scale Set\naz vmss create --name ${vmssName} --resource-group myResourceGroup --location $location --vm-sku $vmSku --instance-count $instanceCount --image UbuntuLTS --admin-username $adminUsername --admin-password $adminPasswordOrKey --vnet-name ${vmssName}vnet --subnet ${vmssName}subnet --load-balancer ${vmssName}lb\n```\n\nPlease replace `myResourceGroup` with your actual resource group name.
